Option Compare Database

Private Sub Cmd_Insertion_DblClick(Cancel As Integer)
    Dim db As Database
    Dim rs As Recordset
    Set db = CurrentDb()
    Set rs = db.OpenRecordset("ETUDIANT", dbOpenDynaset)
    
    rs.AddNew
    rs("Num") = 1211
    rs("Nom") = "GRAGLIA"
    rs("Prenom") = "Valentin"
    rs("Anniversaire") = "12/11/1996"
    rs("Niveau") = "1"
    rs("Groupe") = 5
    rs("Departement") = "Informatique"
    rs.Update
    
End Sub

Private Sub Cmd_Lecture_Click()
    Dim db As Database
    Dim rs As Recordset
    Set db = CurrentDb()
    Set rs = db.OpenRecordset("ETUDIANT")
    
    While Not rs.EOF
        MsgBox rs("Nom")
        rs.MoveNext
    Wend
End Sub

Private Sub Cmd_Modification_MouseMove(Button As Integer, Shift As Integer, X As Single, Y As Single)
    Dim db As Database
    Dim rs As Recordset
    Set db = CurrentDb()
    Set rs = db.OpenRecordset("SELECT * FROM ETUDIANT WHERE Nom='Dupont'", dbOpenDynaset)
    
    If Not rs.EOF Then
        rs.Edit
        rs("Nom") = "Dupond"
        rs.Update
    End If
End Sub

Private Sub Cmd_Suppression_Click()
    Dim db As Database
    Dim rs As Recordset
    Set db = CurrentDb()
    Set rs = db.OpenRecordset("SELECT * FROM ETUDIANT WHERE Nom='DUPRE'", dbOpenDynaset)
    
    If Not rs.EOF Then
        rs.Delete
        rs.MoveNext
    End If
End Sub
